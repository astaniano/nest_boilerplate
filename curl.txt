// create user depricated
curl -i -X POST http://localhost:3000/users -H 'Content-Type: application/json' -d '{"email": "bro@ffffffd.com", "password": "123456"}'
// create user
curl -i -X POST http://localhost:3000/auth/signup -H 'Content-Type: application/json' -d '{"email": "bro@ffff.com", "password": "123456"}'

// signin user
curl -i -X POST http://localhost:3000/auth/signin -H 'Content-Type: application/json' -d '{"email": "bro@ffff.com", "password": "123456"}'

// get all users
curl -i -X GET http://localhost:3000/users

// get role
curl -i -X GET http://localhost:3000/roles/ADMIN

// create role admin
curl -i -X POST http://localhost:3000/roles -H 'Content-Type: application/json' -d '{"value": "ADMIN", "description": "main user"}'
// create role user
curl -i -X POST http://localhost:3000/roles -H 'Content-Type: application/json' -d '{"value": "USER", "description": "regular user"}'
